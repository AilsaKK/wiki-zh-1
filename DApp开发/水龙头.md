<template>
  <div class="page-wrap">
    <div class="header">Thinkium测试链测试币水龙头</div>
    <p>自动发送0.5个测试网TKM到你的测试网地址，最多可以获取5个测试网TKM</p>
    <p>这里的TKM仅能用于测试</p>

    <div class="search-wrap">
      <div class="tab-wrap">
        <div class="tab-item" 
          v-for="(item, index) in tabList" 
          :key="index + item" 
          :class="{'active': selectedIndex === index}" 
        >
          {{item.label}}
        </div>
      </div>
      <div class="search" style="width:388px" v-show="selectedIndex == 1">
        <el-select v-model="coinAddressId" placeholder="请选择" style="width:100%">
          <el-option
            v-for="item in coinList"

        </el-select>
      </div>

        <el-input class="input" v-model="address" placeholder="请输入你的地址..."></el-input>

<script>
import { autoSendTestCoin, autoSendTRCCoin, getCoinAddress } from '../../api/testNetwork.js';
import { isValidTHAddress, toAddress0x } from '../../utils/common';
export default {
  data() {
    return {
      tabList: [
        {label: 'TKM', value: 1},
      ],
      selectedIndex: 0,
      address: '',
      amount: "",
      coinAddressId: '',
      coinList: []
    }
  },
  created() {
  },
  methods: {
    selectClick(index) {
      this.selectedIndex = index
      this.address = ""
    },
    async autoSendTestCoin() {
      let data = {
        toAddress: toAddress0x(this.address),
        type: this.tabList[this.selectedIndex].value
      }
      try{
          let res = await autoSendTestCoin(data);
          this.$message.success('发放成功')
      }catch(err){
         console.log('--err', err)
         this.$message.warning(err && err.msg)
      }
      
    },
    confirmBtn() {
      this.address = this.address.trim();
      if(!this.address) {
        this.$message.warning('请输入地址')
        return false
      }
      if(!isValidTHAddress(this.address)){
        this.$message.warning('请输入正确地址')
        return
      }
      console.log(this.address)
      if(this.selectedIndex == 0) {
        this.autoSendTestCoin()
      }
      if(this.selectedIndex == 1) {
        this.autoSendTRCCoin()
      }
    }
  }
}
</script>
